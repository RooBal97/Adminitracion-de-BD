Table CATEGORIA {
  COD_CATE CHAR(6) [pk]
  NOMBRE VARCHAR(30)
}

Table PRODUCTO {
  ID_PRODUCTO CHAR(6) [pk]
  DESCRIPCION VARCHAR(100)
  PRECIO_VENTA DECIMAL
  STOCK_MINIMO INTEGER
  STOCK_ACTUAL INTEGER
  COD_CATE CHAR(6)
  COD_SEDE CHAR(6)
}

Table DIRECCION {
  ID_DIRECCION CHAR(7) [pk]
  NOMBRE_DISTRITO VARCHAR(20)
  DEPARTAMENTO VARCHAR(20)
  PROVINCIA VARCHAR(20)
  DET_DIRECCION VARCHAR(100)
  REFERENCIA VARCHAR(100)
}

Table SEDE {
  COD_SEDE CHAR(6) [pk]
  NOMBRE_SEDE VARCHAR(20)
  AFORO INT
  NUM_PISOS INT
  ID_DIRECCION CHAR(7)
}

Table CLIENTE {
  ID_CLIENTE CHAR(8) [pk]
  NOMBRES VARCHAR(30)
  APELLIDOS VARCHAR(50)
  TELF CHAR(9)
  EMAIL VARCHAR(30)
}

Table DIREC_CLIENTE {
  ID_DIRECCION CHAR(7)
  ID_CLIENTE CHAR(8)
  FECHA_ASOCIACION DATETIME
}

Table EMPLEADO {
  COD_EMPLE CHAR(6) [pk]
  NOMBRES VARCHAR(30)
  APELLIDOS VARCHAR(50)
  DNI_EMPLE CHAR(8)
  ESTADO_CIVIL VARCHAR (15)
  NIVEL_EDUCA VARCHAR(20)
  TELF CHAR(9)
  EMAIL VARCHAR(30)
  FECHA_INGRESO DATETIME
  COD_CARGO CHAR(6)
  COD_SEDE CHAR(6)
}

Table CARGOS {
  COD_CARGO CHAR(6) [pk]
  NOMBRE_CARGO VARCHAR(20)
  SUELDO_BASICO MONEY
}

Table BOLETA {
  NUM_BOLETA CHAR(10) [pk]
  FECHA_EM DATETIME
  ESTADO_BOLETA VARCHAR (15)
  TOTAL MONEY
  ID_CLIENTE CHAR(8)
  COD_EMPLE CHAR(6)
}

Table DETALLEBOLETA {
  NUM_BOLETA CHAR(10) [pk]
  ID_PRODUCTO CHAR(6)
  CANTIDAD INT
  IMPORTE MONEY
  IGV DECIMAL
}
